### AWS Plugin

```sh
##
# AWS Plugin Installation
#
# @see: https://github.com/iammuhit/vagrant-aws
#
# vagrant box add aws-dummy https://github.com/iammuhit/vagrant-aws/raw/master/dummy.box
# 
vagrant plugin install vagrant-aws

## 
# Create Vagrantfile & Run Virtual Machine
# 
vagrant init -m aws-dummy
vagrant up --provider aws
```

### EC2 â€” Elastic Compute Cloud

```sh
## 
# Create KeyPair
# 
# @see: https://docs.aws.amazon.com/cli/latest/reference/ec2/create-key-pair.html
# 
aws ec2 create-key-pair \
    --key-name deployer-key \
    --query 'KeyMaterial' \
    --output text | tee keys/id_rsa.pem

## 
# Add Rules to Security Group
# 
# @see: https://docs.aws.amazon.com/cli/latest/reference/ec2/create-key-pair.html
# 
aws ec2 authorize-security-group-ingress \
    --group-id default \
    --protocol tcp \
    --port 8000 \
    --cidr 0.0.0.0/0

## 
# Verify Rules of Security Group
# 
# @see: https://docs.aws.amazon.com/cli/latest/reference/ec2/describe-security-groups.html
# 
aws ec2 describe-security-groups --group-name default

## 
# Fetch Available Images (AMIs)
# 
# @see: https://docs.aws.amazon.com/cli/latest/reference/ec2/describe-images.html
# 
aws ec2 describe-images \
  --profile default \
  --region us-east-1 \
  --owners 099720109477 \
  --filters "Name=name,Values=ubuntu/images/hvm-ssd-gp3/ubuntu-noble-24.04-amd64-server-*" \
  --query "Images | sort_by(@, &CreationDate)[-1].[ImageId,Name,CreationDate]" \
  --output text

## 
# Run EC2 Instance
# 
# @see: https://docs.aws.amazon.com/cli/latest/reference/ec2/run-instances.html
# 
aws ec2 run-instances \
    --image-id ami-0b6c6ebed2801a5cb \
    --count 1 \
    --instance-type t2.nano \
    --key-name deployer-key \
    --security-group-ids <value> \
    --user-data file://./user-script.sh

## 
# Assign necessary permissions to the key files for security reasons
# 
chmod 400 keys/id_rsa.pem

## 
# Connecting to EC2 Instance via SSH
# 
ssh -p 22 -i keys/id_rsa.pem ubuntu@<ip-address>
```
